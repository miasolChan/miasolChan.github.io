图片并排

<table frame=void>	<!--使用table标签，且frame=void消除外边框-->
	<tr>		   <!--<tr>一行的内容<\tr>，<td>一个格子的内容<\td>-->
    <td><center><img src="图片地址1"		
                     alt="第一张图片显示不出时显示的文字"
                     height="填数字作为高度"/></center></td>	<!--<center>标签将图片居中-->
    <td><center><img src="图片地址2"
                     alt="第二张图片显示不出时显示的文字"
                     height="高度一致会好看"/></center></td>
    </tr>
</table>
